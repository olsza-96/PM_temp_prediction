def test_Kfeatures(data, dataX, dataY):
    features=SelectKBest(k='all',score_func=f_regression) #selecting all the features and checking which will return the highest performance 
    fit=features.fit(dataX, dataY) #calculastes the f_regression of the features
    print(fit.scores_)
    features_results=fit.transform(dataX)
    print(features_results)

    chart = sns.barplot(x=data.drop(columns = ['pm']).columns, y =fit.scores_)
    chart.set_xticklabels(chart.get_xticklabels(), rotation=45, horizontalalignment='right')
    plt.title('Best feature scores for the dataset')
    plt.xlabel('Feature')
    plt.ylabel('Score')
    plt.show()
